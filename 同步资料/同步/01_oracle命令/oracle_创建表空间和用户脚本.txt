--创建表空间
CREATE TABLESPACE DLZSB
    LOGGING 
    DATAFILE 'E:\MyDatabase\oracle\DLZSB.ora' SIZE 50M REUSE 
    AUTOEXTEND 
    ON NEXT  10M MAXSIZE UNLIMITED EXTENT MANAGEMENT LOCAL;


---子系统
CREATE USER DLZSB  PROFILE "DEFAULT" IDENTIFIED BY 
    DLZSB DEFAULT 
    TABLESPACE DLZSB TEMPORARY 
    TABLESPACE "TEMP" ACCOUNT UNLOCK;

GRANT UNLIMITED TABLESPACE TO DLZSB;
GRANT "CONNECT"    TO DLZSB;
GRANT "RESOURCE"   TO DLZSB;
GRANT CREATE VIEW  TO DLZSB;
GRANT"DBA" TO DLZSB WITH ADMIN OPTION;

GRANT SELECT ON "SYS"."V_$SESSION" TO DLZSB;
GRANT SELECT ON "SYS"."V_$SESSTAT" TO DLZSB;
GRANT SELECT ON "SYS"."V_$STATNAME" TO DLZSB;
